-- a) 
CREATE TABLE DOG_BREEDS (BREED VARCHAR2(50));
CREATE TABLE CUSTOMERS (C_ID NUMBER, F_NAME VARCHAR2(50), L_NAME VARCHAR2(50), DOB DATE);
CREATE TABLE DOGS (DOG_ID NUMBER, DOG_NAME VARCHAR2(50), DOG_BREED VARCHAR2(50), C_ID NUMBER);
CREATE TABLE STORES (STORE_ID NUMBER, STORE_AREA VARCHAR2(50));
CREATE TABLE SERVICE_HISTORY (DOG_ID NUMBER, STORE_ID NUMBER, SERVICE_ID NUMBER, FINISHED CHAR(1), MESSAGE VARCHAR2(300));
CREATE TABLE SERVICE_HISTORY_DETAIL (SERVICE_ID NUMBER, START_TIME TIMESTAMP, END_TIME TIMESTAMP, SERVICE_NAME VARCHAR2(50));
CREATE TABLE SERVICES (SERVICE_NAME VARCHAR2(50), PRICE NUMBER(4, 2));

-- b)
ALTER TABLE DOG_BREEDS
ADD CONSTRAINT PK_DOG_BREEDS PRIMARY KEY (BREED);
ALTER TABLE DOGS
ADD CONSTRAINT PK_DOGS PRIMARY KEY (DOG_ID);
ALTER TABLE CUSTOMERS
ADD CONSTRAINT PK_CUSTOMERS PRIMARY KEY (C_ID);
ALTER TABLE STORES
ADD CONSTRAINT PK_STORES PRIMARY KEY (STORE_ID);
ALTER TABLE SERVICE_HISTORY
ADD CONSTRAINT PK_SERVICE_HISTORY PRIMARY KEY (SERVICE_ID);
ALTER TABLE SERVICE_HISTORY_DETAIL
ADD CONSTRAINT PK_SHD PRIMARY KEY (SERVICE_ID);
ALTER TABLE SERVICES
ADD CONSTRAINT PK_SERVICES PRIMARY KEY (SERVICE_NAME);
ALTER TABLE DOGS ADD CONSTRAINT FK_DOG_BREED FOREIGN KEY (DOG_BREED)
REFERENCES DOG_BREEDS (BREED);
ALTER TABLE DOGS ADD CONSTRAINT FK_C_ID FOREIGN KEY (C_ID)
REFERENCES CUSTOMERS (C_ID);
ALTER TABLE SERVICE_HISTORY ADD CONSTRAINT FK_SH_DOG_ID FOREIGN KEY (DOG_ID) 
REFERENCES DOGS (DOG_ID);
ALTER TABLE SERVICE_HISTORY ADD CONSTRAINT FK_SH_STORE_ID FOREIGN KEY (STORE_ID)
REFERENCES STORES (STORE_ID);
ALTER TABLE SERVICE_HISTORY_DETAIL ADD CONSTRAINT FK_SHD_SERVICE_ID FOREIGN KEY (SERVICE_ID)
REFERENCES SERVICE_HISTORY (SERVICE_ID);
ALTER TABLE SERVICE_HISTORY_DETAIL ADD CONSTRAINT FK_SERVICE_NAME FOREIGN KEY (SERVICE_NAME)
REFERENCES SERVICES (SERVICE_NAME);
ALTER TABLE DOGS MODIFY DOG_NAME VARCHAR2(50) NOT NULL;
ALTER TABLE CUSTOMERS MODIFY DOB DATE NOT NULL;
ALTER TABLE SERVICES MODIFY PRICE NUMBER(4, 2) NOT NULL;
ALTER TABLE SERVICE_HISTORY ADD CONSTRAINT CK_FINISHED
CHECK ((FINISHED ='T') OR (FINISHED = 'F'));
ALTER TABLE CUSTOMERS ADD CONSTRAINT CK_DOB
CHECK (DOB <= DATE '1998-12-31');
ALTER TABLE SERVICE_HISTORY_DETAIL ADD CONSTRAINT CK_START_TIME_END_TIME
CHECK (START_TIME < END_TIME);
ALTER TABLE SERVICE_HISTORY_DETAIL ADD CONSTRAINT CK_SERVICE_DATE
CHECK (END_TIME < DATE '2018-1-1');
